#!/usr/bin/python3

nb=input("donner le nombre des etudiants : ")
nb=int(nb)

matiere=["math","reseau","cloud"]

note=["ds","exam","tp"]

fnom=fopen("nom.txt","w")

fmath=fopen("math.txt","w")

freseau=fopen("reseau.txt","w")

fcloud=fopen("cloud.txt","w")

tableau=[fmath,freseau,fcloud]

for i in range(nb):
	
fnom.write(input("donnez le nom"))

	fnom.write("\n")
	
for j in matiere:

		for k in note:

			print("donner la note du:",k,"de",j)
	
		tableau[matiere.index(j)].write(input())
		
	tableau[matiere.index(j)].write(" ")
	
	tableau[matiere.index(j)].write("\n")
		
	
fnom.close()
fmath.close()
freseau.close()
fcloud.close()
